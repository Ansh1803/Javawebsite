class y {
            public static void main(String[] args) {
                int[] array = {1, 2, 3, 4, 5};
                int k = 2; 
                int n = array.length;
                
                
                k = k % n;
        
         
                int[] rotatedArray = new int[n];
        
                for (int i = 0; i < n; i++) {
                    rotatedArray[(i + k) % n] = array[i];
                }
        
              
                for (int i = 0; i < n; i++) {
                    System.out.print(rotatedArray[i] + " ");
                }
            }
        }
