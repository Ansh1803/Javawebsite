class d1 {
            public static void main(String[] args) {
                int[] array = {1, 2, 3, 4, 5, 2, 3, 6, 7, 8, 1};
                boolean[] seen = new boolean[100]; 
        
                for (int num : array) {
                    if (num >= 0 && num < 100) {
                        if (seen[num]) {
                            System.out.println(num);
                        } else {
                            seen[num] = true;
                        }
                    }
                }
            }
        }
