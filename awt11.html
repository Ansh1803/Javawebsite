import java.awt.*;
import java.awt.event.*;

class CustomChoiceApp {
    public static void main(String[] args) {
        final Frame frame = new Frame("Custom Choice AWT Application");
        frame.setLayout(new GridLayout(2, 1));

        Label label = new Label("Select an option:", Label.CENTER);
        final Choice choice = new Choice();
        
        choice.add("Option 1");
        choice.add("Option 2");
        choice.add("Option 3");

        Button button = new Button("Show Selection");
        button.addActionListener(new ActionListener() {
            public void actionPerformed(ActionEvent e) {
                String selected = choice.getSelectedItem();
                System.out.println("Selected: " + selected);
            }
        });

        frame.add(label);
        frame.add(choice);
        frame.add(button);

        frame.setSize(300, 150);
        frame.addWindowListener(new WindowAdapter() {
            public void windowClosing(WindowEvent we) {
                System.exit(0);
            }
        });

        frame.setVisible(true);
    }
}
