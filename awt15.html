import java.awt.*;
import java.awt.event.*;

class mApp {
    public static void main(String[] args) {
        final Frame frame = new Frame("Custom List AWT Application");
        frame.setLayout(new GridLayout(3, 1));

        Label label = new Label("Select an option:", Label.CENTER);
        final List list = new List(4); 

        list.add("Option 1");
        list.add("Option 2");
        list.add("Option 3");
        list.add("Option 4");
        list.add("Option 5");

        Button button = new Button("Show Selection");
        button.addActionListener(new ActionListener() {
            public void actionPerformed(ActionEvent e) {
                String[] selectedItems = list.getSelectedItems();
                if (selectedItems.length > 0) {
                    StringBuilder selected = new StringBuilder();
                    for (String item : selectedItems) {
                        selected.append(item).append(", ");
                    }
                    if (selected.length() > 0) {
                        selected.setLength(selected.length() - 2);
                    }
                    System.out.println("Selected: " + selected.toString());
                } else {
                    System.out.println("No option selected.");
                }
            }
        });

        frame.add(label);
        frame.add(list);
        frame.add(button);

        frame.setSize(300, 200);
        

        frame.setVisible(true);
    }
}
