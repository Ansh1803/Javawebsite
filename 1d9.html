class  c12{
            public static void main(String[] args) {
                int[] array = {3, 5, 7, 2, 8, 6};
                int largest = array[0];
                for (int i = 1; i < array.length; i++) {
                    if (array[i] > largest) {
                        largest = array[i];
                    }
                }
                System.out.println(largest);
            }
        }
