import java.util.Arrays;
        import java.util.List;
        
        public class PositiveNumberException {
            public static void main(String[] args) {
                try {
                    checkPositiveNumbers(Arrays.asList(-1, -2, 3));
                } catch (Exception e) {
                    System.out.println("Caught an exception: " + e.getMessage());
                }
            }
        
            public static void checkPositiveNumbers(List numbers) throws Exception {
                for (int number : numbers) {
                    if (number > 0) {
                        throw new Exception("List contains positive numbers.");
                    }
                }
            }
        }

Output:

Caught an exception: List contains positive numbers.
