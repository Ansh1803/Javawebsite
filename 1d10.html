class s2{
            public static void main(String[] args) {
                int[] array = {3, 5, 7, 2, 8, 6};
                int smallest = array[0];
                for (int i = 1; i < array.length; i++) {
                    if (array[i] < smallest) {
                        smallest = array[i];
                    }
                }
                System.out.println(smallest);
            }
        }
