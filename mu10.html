class NamedThread extends Thread {
            public NamedThread(String name) {
                super(name);
            }
        
            public void run() {
                System.out.println("Thread: " + getName() + " with priority: " + getPriority());
            }
        }
        
        public class NamedThreadExample {
            public static void main(String[] args) {
                NamedThread thread1 = new NamedThread("HighPriorityThread");
                NamedThread thread2 = new NamedThread("LowPriorityThread");
                thread1.setPriority(Thread.MAX_PRIORITY);
                thread2.setPriority(Thread.MIN_PRIORITY);
                thread1.start();
                thread2.start();
            }
        }

Output:

Thread: HighPriorityThread with priority: 10 Thread: LowPriorityThread with priority: 1
