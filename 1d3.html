class Left {
            public static void main(String[] args) {
                
                int[] array = {1, 2, 3, 4, 5};
                int d = 2;
        
                
                leftRotate(array, d);
        
                
                System.out.println("Rotated Array:");
                for (int num : array) {
                    System.out.print(num + " ");
                }
            }
        
            public static void leftRotate(int[] array, int d) {
                int n = array.length;
               
                d = d % n; 
        
                
                int[] temp = new int[n];
        
                
                for (int i = 0; i < n; i++) {
                    temp[i] = array[(i + d) % n];
                }
        
               
                for (int i = 0; i < n; i++) {
                    array[i] = temp[i];
                }
            }
        }
