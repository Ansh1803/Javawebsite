class SortThread extends Thread {
            private int[] arr;
            private int start, end;
        
            public SortThread(int[] arr, int start, int end) {
                this.arr = arr;
                this.start = start;
                this.end = end;
            }
        
            public void run() {
                java.util.Arrays.sort(arr, start, end);
            }
        }
        
        public class MultiThreadSort {
            public static void main(String[] args) {
                int[] arr = {5, 3, 8, 1, 2, 7, 4, 6};
                SortThread t1 = new SortThread(arr, 0, arr.length / 2);
                SortThread t2 = new SortThread(arr, arr.length / 2, arr.length);
                
                t1.start();
                t2.start();
                
                try {
                    t1.join();
                    t2.join();
                } catch (InterruptedException e) {
                    e.printStackTrace();
                }
                
                java.util.Arrays.sort(arr); // Final sort
                System.out.println(java.util.Arrays.toString(arr));
            }
        }

Output:

[1, 2, 3, 4, 5, 6, 7, 8]
