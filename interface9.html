interface Searchable {
            void search(String keyword);
        }
        
        class Document implements Searchable {
            
            public void search(String keyword) {
                System.out.println("Searching for '" + keyword + "' in Document.");
            }
        }
        
        class WebPage implements Searchable {
        
            public void search(String keyword) {
                System.out.println("Searching for '" + keyword + "' in WebPage.");
            }
        }
        
        public class SearchableExample {
            public static void main(String[] args) {
                Searchable document = new Document();
                Searchable webpage = new WebPage();
        
                document.search("Java");
                webpage.search("Interface");
            }
        }

Output:

Searching for 'Java' in Document. Searching for 'Interface' in WebPage.
