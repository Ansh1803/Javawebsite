class x3{
            public static void main(String[] args) {
                int[][] matrix = {
                    {1, 2, 3},
                    {4, 5, 6},
                    {7, 8, 9}
                };
        
                System.out.println("The upper triangular matrix is:");
                for (int i = 0; i < matrix.length; i++) {
                    for (int j = 0; j < matrix[i].length; j++) {
                        if (i <= j) {
                            System.out.print(matrix[i][j] + " ");
                        } else {
                            System.out.print("0 ");
                        }
                    }
                    System.out.println();
                }
            }
        }
