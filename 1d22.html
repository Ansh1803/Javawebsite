class x1 {
            public static void main(String[] args) {
                int[] array = {5, 2, 8, 3, 7, 6, 1, 4};
        
                System.out.print("Even numbers: ");
                for (int num : array) {
                    if (num % 2 == 0) {
                        System.out.print(num + " ");
                    }
                }
        
                System.out.print("\nOdd numbers: ");
                for (int num : array) {
                    if (num % 2 != 0) {
                        System.out.print(num + " ");
                    }
                }
            }
        }
