class g1 {
            public static void main(String[] args) {
                int[] array = {5, 2, 8, 1, 3, 7, 6};
        
                int largest = array[0];
        
                for (int i = 1; i < array.length; i++) {
                    if (array[i] > largest) {
                        largest = array[i];
                    }
                }
        
                System.out.println("The largest number is: " + largest);
            }
        }
