import java.util.Arrays;

        class MatrixFrequency {
            public static void main(String[] args) {
                int[][] matrix = {
                    {1, 2, 3},
                    {4, 5, 6},
                    {7, 8, 9}
                };
        
                int evenCount = 0;
                int oddCount = 0;
        
                for (int[] row : matrix) {
                    for (int num : row) {
                        if (num % 2 == 0) {
                            evenCount++;
                        } else {
                            oddCount++;
                        }
                    }
                }
        
                System.out.println("Even numbers: " + evenCount);
                System.out.println("Odd numbers: " + oddCount);
            }
        }
