public class NestedTryExample {
            public static void main(String[] args) {
                try {
                    try {
                        int result = 10 / 0;
                    } catch (ArithmeticException e) {
                        System.out.println("Caught inner exception: " + e.getMessage());
                    }
                    String str = null;
                    System.out.println(str.length());
                } catch (NullPointerException e) {
                    System.out.println("Caught outer exception: " + e.getMessage());
                }
            }
        }

Output:

Caught inner exception: / by zero Caught outer exception: Cannot invoke "String.length()" because "str" is null
