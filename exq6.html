import java.util.HashSet;
        import java.util.Set;
        
        public class DuplicateNumberException {
            public static void main(String[] args) {
                try {
                    checkDuplicates(new int[]{1, 2, 3, 2});
                } catch (Exception e) {
                    System.out.println("Caught an exception: " + e.getMessage());
                }
            }
        
            public static void checkDuplicates(int[] numbers) throws Exception {
                Set seen = new HashSet<>();
                for (int number : numbers) {
                    if (!seen.add(number)) {
                        throw new Exception("Duplicate number found: " + number);
                    }
                }
            }
        }

Output:

Caught an exception: Duplicate number found: 2
