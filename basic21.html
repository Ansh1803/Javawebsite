public class SwapArray {
    public static void main(String[] args) {
        int[] arr = {1, 2, 3, 4, 5};
        int[] newArr = arr.clone();
        int temp = newArr[0];
        newArr[0] = newArr[newArr.length - 1];
        newArr[newArr.length - 1] = temp;
        for (int num : newArr) {
            System.out.print(num + " ");
        }
    }
}
