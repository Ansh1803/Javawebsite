import java.awt.*;
import java.awt.event.*;

class pApp {
    public static void main(String[] args) {
        final Frame frame = new Frame("Checkbox Group AWT Application");
        frame.setLayout(new FlowLayout());

        Label label = new Label("Select an option:");
       final CheckboxGroup group = new CheckboxGroup();

        Checkbox checkbox1 = new Checkbox("Option 1", group, false);
        Checkbox checkbox2 = new Checkbox("Option 2", group, false);
        Checkbox checkbox3 = new Checkbox("Option 3", group, false);

        Button button = new Button("Show Selection");
        button.addActionListener(new ActionListener() {
            public void actionPerformed(ActionEvent e) {
                Checkbox selected = group.getSelectedCheckbox();
                if (selected != null) {
                    System.out.println("Selected: " + selected.getLabel());
                } else {
                    System.out.println("No option selected.");
                }
            }
        });

        frame.add(label);
        frame.add(checkbox1);
        frame.add(checkbox2);
        frame.add(checkbox3);
        frame.add(button);

        frame.setSize(300, 150);
        frame.addWindowListener(new WindowAdapter() {
            public void windowClosing(WindowEvent we) {
                System.exit(0);
            }
        });

        frame.setVisible(true);
    }
}
